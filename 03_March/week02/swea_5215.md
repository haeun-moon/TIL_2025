import java.util.Scanner;

public class swea_5215 {

    static int [][] map;
    static int N;
    static int K;
    static int max;
    static int sumK;
    static int sumT;
    static int tc;

    public static void main(String[] args) {
        Scanner sc = new Scanner (System.in);

        int T = sc.nextInt();

        for (tc = 1; tc <= T; tc++) {

            N = sc.nextInt();
            K = sc.nextInt();

            map = new int [N][2];

            for (int n = 0; n < N; n++) {
                map[n][0] = sc.nextInt();
                map[n][1] = sc.nextInt();
            }

            max = 0;
            comb(0,0);

        }//tc
    }//main

    public static void comb(int idx, int sidx) {

        if (idx == N) {
            max = Math.max(max, sumT);
            System.out.println("#" + tc + " " + max);
            return;
        }

        if (sidx == N) {
            max = Math.max(max, sumT);
            sumT = 0;
            sumK = 0;
            idx++;
            sidx = 0;
            System.out.println("IF - idx: " + idx + "/ sidx: " + sidx + "/ max: " + max + "/ sumT: " + sumT + "/ sumK: " + sumK);
            comb(idx, sidx);
        } else if (sumK > K) {
            sumT -= map[sidx-1][0];
            sumK -= map[sidx-1][1];
            sidx++;
            System.out.println("ELSE IF - idx: " + idx + "/ sidx: " + sidx + "/ max: " + max + "/ sumT: " + sumT + "/ sumK: " + sumK);
            comb(idx, sidx);
        } else {

            sumK += map[sidx][1];
            sumT += map[sidx][0];
            System.out.println("ELSE - idx: " + idx + "/ sidx: " + sidx + "/ max: " + max + "/ sumT: " + sumT + "/ sumK: " + sumK);
            comb(idx, sidx + 1);

        }
    }
}
